                            using namespace std;
                            
                            enum class Direction { STOP = 0, LEFT, RIGHT, UP, DOWN };
                            
                            struct GameSettings {
                                int width;
                                int height;
                                int speed;
                                int score;
                                bool game_over;
                                Direction dir;
                                vector tailX;
                                vector tailY;
                                int fruitX;
                                int fruitY;
                            };
                            
                            struct Score {
                                string name;
                                int score;
                            };
                            
                            enum class Difficulty { EASY, MEDIUM, HARD };
                            
                            Difficulty GetDifficultyFromUser() {
                                char choice;
                                cout << "Select difficulty:" << endl;
                                cout << "1. Easy" << endl;
                                cout << "2. Medium" << endl;
                                cout << "3. Hard" << endl;
                                cout << "Enter choice: ";
                                cin >> choice;
                                switch (choice) {
                                    case '1':
                                        return Difficulty::EASY;
                                    case '2':
                                        return Difficulty::MEDIUM;
                                    case '3':
                                        return Difficulty::HARD;
                                    default:
                                        cout << "Invalid choice. Setting difficulty to Medium." << endl;
                                        return Difficulty::MEDIUM;
                                }
                            }
                            
                            void DrawSnakeMenu() {
                                cout << "  SSSSSS  EEEEEEE  RRRRRR  PPPPPP" << endl;
                                cout << " S        E        R    R  P    P" << endl;
                                cout << "  SSSSSS  EEEE     RRRRRR  PPPPPP" << endl;
                                cout << "        S E        R R     P" << endl;
                                cout << "  SSSSSS  EEEEEEE  R  R    P" << endl;
                            }
                            
                            void InitializeGame(GameSettings& game, Difficulty difficulty) {
                                game.width = 30;
                                game.height = 15;
                                game.speed = (difficulty == Difficulty::EASY) ? 50 :
                                            ((difficulty == Difficulty::MEDIUM) ? 100 : 250);
                                game.score = 0;
                                game.game_over = false;
                                game.dir = Direction::RIGHT;
                                game.tailX = {3, 2, 1};
                                game.tailY = {1, 1, 1};
                                game.fruitX = rand() % game.width;
                                game.fruitY = rand() % game.height;
                            }
                            
                            void DrawGame(const GameSettings& game) {
                                HANDLE hConsole = GetStdHandle(STD_OUTPUT_HANDLE);
                                COORD cursorPos;
                                cursorPos.X = 0;
                                cursorPos.Y = 0;
                                SetConsoleCursorPosition(hConsole, cursorPos);
                            
                                // Cambia el fondo de la consola a rojo y el color del texto a negro
                                SetConsoleTextAttribute(hConsole, BACKGROUND_RED | FOREGROUND_INTENSITY | FOREGROUND_GREEN | FOREGROUND_BLUE);
                            
                                for (int i = 0; i < game.height; i++) {
                                    for (int j = 0; j < game.width; j++) {
                                        if (j == 0 || j == game.width - 1)
                                            cout << "#";
                                        else if (i == game.fruitY && j == game.fruitX)
                                            cout << "F";
                                        else {
                                            bool print = false;
                                            for (size_t k = 0; k < game.tailX.size(); k++) {
                                                if (game.tailX[k] == j && game.tailY[k] == i) {
                                                    cout << "o";
                                                    print = true;
                                                }
                                            }
                                            if (!print)
                                                cout << " ";
                                        }
                                    }
                                    cout << endl;
                                }
                            
                                cout << "Score:" << game.score << endl;
                            }
                            
                            void HandleInput(GameSettings& game) {
                                if (_kbhit()) {
                                    switch (_getch()) {
                                        case 75:  // Flecha Izquierda
                                            if (game.dir != Direction::RIGHT)
                                                game.dir = Direction::LEFT;
                                            break;
                                        case 77:  // Flecha Derecha
                                            if (game.dir != Direction::LEFT)
                                                game.dir = Direction::RIGHT;
                                            break;
                                        case 72:  // Flecha Arriba
                                            if (game.dir != Direction::DOWN)
                                                game.dir = Direction::UP;
                                            break;
                                        case 80:  // Flecha Abajo
                                            if (game.dir != Direction::UP)
                                                game.dir = Direction::DOWN;
                                            break;
                                        case 27:  // Tecla Esc (Salir)
                                            game.game_over = true;
                                            break;
                                    }
                                }
                            }
                            
                            void Logic(GameSettings& game) {
                                int prevX = game.tailX[0];
                                int prevY = game.tailY[0];
                                int prev2X, prev2Y;
                                game.tailX[0] = game.tailX[0] + (game.dir == Direction::RIGHT ? 1 : (game.dir == Direction::LEFT ? -1 : 0));
                                game.tailY[0] = game.tailY[0] + (game.dir == Direction::DOWN ? 1 : (game.dir == Direction::UP ? -1 : 0));
                            
                                if (game.tailX[0] >= game.width) game.tailX[0] = 0;
                                if (game.tailX[0] < 0) game.tailX[0] = game.width - 1;
                                if (game.tailY[0] >= game.height) game.tailY[0] = 0;
                                if (game.tailY[0] < 0) game.tailY[0] = game.height - 1;
                            
                                for (size_t i = 1; i < game.tailX.size(); i++) {
                                    prev2X = game.tailX[i];
                                    prev2Y = game.tailY[i];
                                    game.tailX[i] = prevX;
                                    game.tailY[i] = prevY;
                                    prevX = prev2X;
                                    prevY = prev2Y;
                                }
                            
                                for (size_t i = 1; i < game.tailX.size(); i++) {
                                    if (game.tailX[i] == game.tailX[0] && game.tailY[i] == game.tailY[0])
                                        game.game_over = true;
                                }
                            
                                if (game.tailX[0] == game.fruitX && game.tailY[0] == game.fruitY) {
                                    game.score += 10;
                                    game.fruitX = rand() % game.width;
                                    game.fruitY = rand() % game.height;
                                    game.tailX.push_back(prevX);
                                    game.tailY.push_back(prevY);
                                }
                            }
                            
                            
                            int main() {
                                Difficulty difficulty = GetDifficultyFromUser();
                                GameSettings game;
                                InitializeGame(game, difficulty);
                            
                                // Oculta el cursor de la consola
                                HANDLE hConsole = GetStdHandle(STD_OUTPUT_HANDLE);
                                CONSOLE_CURSOR_INFO cursorInfo;
                                cursorInfo.dwSize = 1;
                                cursorInfo.bVisible = false;
                                SetConsoleCursorInfo(hConsole, &cursorInfo);
                            
                                // Menú de inicio con la serpiente
                                cout << "Welcome to Snake Game!" << endl;
                                DrawSnakeMenu();
                                cout << "Select difficulty:" << endl;
                            
                                while (!game.game_over) {
                                    DrawGame(game);
                                    HandleInput(game);
                                    Logic(game);
                            
                                    // Lógica de velocidad...
                                }
                            
                                // Restaura la visibilidad del cursor al salir del juego
                                cursorInfo.bVisible = true;
                                SetConsoleCursorInfo(hConsole, &cursorInfo);
                            
                                cout << "Game Over! Final Score: " << game.score << endl;
                                return 0;
                            }
